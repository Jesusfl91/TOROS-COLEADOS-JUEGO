import { TonClient, WalletContractV4, internal } from '@ton/ton'; import { mnemonicToPrivateKey } from 'ton-crypto'; import admin from 'firebase-admin'; // Inicializamos Firebase Admin (Solo se inicializa una vez) if (!admin.apps.length) { admin.initializeApp({ credential: admin.credential.cert(JSON.parse(process.env.FIREBASE_SERVICE_ACCOUNT_KEY)) }); } const db = admin.firestore(); export default async function handler(req, res) { // Validar que solo se acepten peticiones POST if (req.method !== 'POST') { return res.status(405).json({ error: 'Método no permitido. Usa POST.' }); } const { userId } = req.body; // Configuración del juego const TASA_CONVERSION = 1000; // 1000 puntos = 1 TON const PUNTOS_MINIMOS = 100; try { const userRef = db.collection('usuarios').doc(userId); // PASO 1: Transacción Atómica de Firebase para evitar Doble Gasto const txResult = await db.runTransaction(async (transaction) => { const doc = await transaction.get(userRef); if (!doc.exists) throw new Error('Usuario no encontrado en los registros.'); const data = doc.data(); const puntosActuales = data.puntos_acumulados || 0; const walletUsuario = data.wallet; if (!walletUsuario) throw new Error('El usuario no tiene una wallet registrada.'); if (puntosActuales < PUNTOS_MINIMOS) throw new Error('No hay puntos suficientes para reclamar.'); // Calcular el monto en TON a enviar const montoTon = (puntosActuales / TASA_CONVERSION).toString(); // Restamos los puntos en Firebase INMEDIATAMENTE transaction.update(userRef, { puntos_acumulados: 0 }); return { walletUsuario, montoTon, puntosDeducidos: puntosActuales }; }); // PASO 2: Inicializar cliente TON const client = new TonClient({ endpoint: 'https://toncenter.com/api/v2/jsonRPC', apiKey: process.env.TONCENTER_API_KEY }); // Obtener la Semilla desde la Variable de Entorno const mnemonic = process.env.SEMILLA_MANGA; if (!mnemonic) throw new Error('Error interno: Billetera de la manga no configurada.'); // Generar llaves y abrir el contrato V4 const keyPair = await mnemonicToPrivateKey(mnemonic.split(' ')); const walletManga = WalletContractV4.create({ workchain: 0, publicKey: keyPair.publicKey }); const contract = client.open(walletManga); // Verificar el secuenciador actual const seqno = await contract.getSeqno(); // Enviar la transferencia de TON await contract.sendTransfer({ seqno, secretKey: keyPair.secretKey, messages: [ internal({ to: txResult.walletUsuario, value: txResult.montoTon, body: 'Pago de Coleada - Recompensa del Juego', bounce: false, }) ] }); // Respuesta exitosa return res.status(200).json({ success: true, message: 'Pago enviado con éxito', tonEnviado: txResult.montoTon }); } catch (error) { console.error('Error procesando el pago en el cajero:', error); return res.status(500).json({ error: 'Error en el cajero, contacta al Juez de Mesa.', details: error.message }); } }
